\subsubsection{CodeBinaire}

\begin{tad}
    \tadNom{CodeBinaire}
    \tadDependances{Statistiques, Octet}
    \begin{tadOperations}{CodeBinaire}
        \tadOperation{creerCodeBinaire}{Statistiques}{CodeBinaire}
        \tadOperation{ajouterBit}{CodeBinaire x Byte}{CodeBinaire}
        \tadOperation{obtenirLongueur}{CodeBinaire}{\naturel}
        \tadOperationAvecPreconditions{obtenirBit}{CodeBinaire x \naturel}{Byte}
        \tadOperation{concatener}{CodeBinaire x CodeBinaire}{CodeBinaire}
    \end{tadOperations}
    \begin{tadSemantiques}{CodeBinaire}
        \tadSemantique{creerCodeBinaire}{Crée un code binaire à partir des statistiques fournies.}
        \tadSemantique{ajouterBit}{Ajoute un bit au code binaire.}
        \tadSemantique{obtenirLongueur}{Renvoie la longueur du code binaire (le nombre de bits).}
        \tadSemantique{obtenirBit}{Renvoie le bit à la position donnée dans le code binaire.}
        \tadSemantique{concatener}{Concatène deux codes binaires.}
      \end{tadSemantiques}    
    \begin{tadAxiomes}
        \tadAxiome{obtenirLongueur(concatener(cb1,cb2))=obtenirLongueur(cb1)+obtenirLongueur(cb2)}
    \end{tadAxiomes}
    \begin{tadPreconditions}{obtenirBit}
        \tadPrecondition{obtenirBit(codeBinaire, position)}{0<=position<obtenirLongueur(codeBinaire)}
        \end{tadPreconditions}
\end{tad}
