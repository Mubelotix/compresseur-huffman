
\subsubsection{TAD File de priorité}

\begin{tad}
  \tadNom{FileDePriorite}
  \tadDependances{ArbreDeHuffman, Naturel}
  
  \begin{tadOperations}{fileDePrioriteVide}
    \tadOperation{fileDePrioriteVide}{}{FileDePriorité}
    \tadOperation{ajouterArbre}{ArbreDeHuffman X FileDePriorité}{FileDePriorité}
    \tadOperationAvecPreconditions{defilerArbre}{FileDePriorité}{Arbre}
    \tadOperation{tailleFile}{FileDePriorité}{Naturel}
  \end{tadOperations}

  \begin{tadSemantiques}{fileDePrioriteVide}
    \tadSemantique{fileDePrioriteVide}{Crée une file de priorité vide.}
    \tadSemantique{ajouterArbre}{Ajoute un arbre de Huffman à son emplacement respectif dans une file de priorité.}
    \tadSemantique{defilerArbre}{Défile l'arbre en tête d'une file de priorité.}
    \tadSemantique{tailleFile}{Renvoie le nombre d'arbre de Huffman présents dans une file de priorité.}
  \end{tadSemantiques}

  \begin{tadAxiomes}
    \tadAxiome{ajouterArbre(defilerArbre(f),f) = f}
    \tadAxiome{tailleFile(fileDePrioriteVide()) = 0}
	  \tadAxiome{tailleFile(ajouterArbre(A,f)) = tailleFile(f) + 1}
  \end{tadAxiomes}

  \begin{tadPreconditions}{defilerArbre(f)}
    \tadPrecondition{defilerArbre(f)}{tailleFile(f) > 0}
  \end{tadPreconditions}
\end{tad}
